```javascript
// backend/routes/dogs.js
router.get('/search', async (req, res) => {
  const { keyword, breed, minAge, maxAge, location } = req.query;
  
  let sql = 'SELECT * FROM dogs WHERE 1=1';
  const params = [];
  
  if (keyword) {
    sql += ' AND (name LIKE ? OR breed LIKE ?)';
    params.push(`%${keyword}%`, `%${keyword}%`);
  }
  
  if (breed) {
    sql += ' AND breed IN (?)';
    params.push(breed.split(','));
  }
  
  if (location) {
    sql += ' AND location LIKE ?';
    params.push(`%${location}%`);
  }
  
  const dogs = await db.query(sql, params);
  res.json(dogs);
});
```
